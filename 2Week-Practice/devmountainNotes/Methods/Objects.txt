Objects - are a common way to show related key-value pairs
Object Syntax: 
const dog = {
  name: 'Freida',
  color: 'brown/black',
  hunger: 40,
  mood: 'feisty',
  age: 9,
};
dog.name = ‘Freida’
dog[‘name’] = ‘Freida’

CLASSES and OBJECTS
class Dog {
    constructor(name, color){
      this.name = name;
      this.color = color;
    }

let dog1 = new Dog('Freida', 'brown');

class Puppy extends Dog {
  constructor(name, color, trainingLevel) {
    super(name, color)

  this.trainingLevel = trainingLevel
  }
